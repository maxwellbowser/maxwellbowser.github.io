def calc_commission_rate(x):
    """ Return the commission rate based on the table:
    0-90% = 2%
    91-99% = 3%
    >= 100 = 4%
    """
    if x <= .90:
        return .02
    if x <= .99:
        return .03
    else:
        return .04

# Adding in the commission column!
df['Commission_Rate'] = df['Pct_To_Target'].apply(calc_commission_rate)
df['Commission_Amount'] = df['Commission_Rate'] * df['Sales']

# Making the loop!

total_stats = []

# How many times to run the simulation
N_SIMUL = 1000

for i in range(N_SIMUL):

    # Running the random sampling from np

    # Getting random sales goal from the given distibution, with corresponding chances
    random_stats = np.random.choice(sales_target_values, N_REP, p=sales_target_prob)
    # Getting the odds that the sales rep actually hits their goals 
    random_pcts = np.random.normal(AVG, STD, N_REP).round(2)

    total_df = pd.DataFrame(index = range(N_REP), data={"target_odds": random_pcts, "target_sales": random_stats})

    # Simulating what the sales reps will actually bring in
    total_df["sales"] = total_df.target_odds * total_df.target_sales

    # Figuring out how much the reps make from commissions
    total_df["commission_rate"] = total_df["target_odds"].apply(calc_commission_rate)
    total_df["commission_amount"] = total_df["commission_rate"] * total_df.sales

    total_stats.append([total_df.sales.sum().round(0), 
                        total_df.commission_amount.sum().round(0), 
                        total_df.target_sales.sum().round(0)])
