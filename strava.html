# Previously did json.dump() so I don't leak my api creds lol
with open("Strava-Creds.json", "r") as f:
    payload = f.read()

# This converts it from the read str type to dict
payload = json.loads(payload)
In[]:
print("Requesting Token...\n")
res = requests.post(auth_url, data=payload, verify=False)
access_token = res.json()["access_token"]

header = {"Authorization": "Bearer " + access_token}
param = {"per_page": 200, "page": 1}
my_dataset = requests.get(activites_url, headers=header, params=param).json()

activities = pd.json_normalize(my_dataset)

print(activities.head())

to_datetime = pd.to_datetime(activities["start_date_local"])

distance = cool_data["distance"] / (1000 * 1.609)
cool_data.loc[:, "distance"] = distance

hours = cool_data["start_datetime_local"].dt.hour + (cool_data["start_datetime_local"].dt.minute / 60)

hours = round(hours, 3)

# Doing thing for coloring according to exercise type, tbh this is so much easier in R :(
activity_colors = {
    "Run": "black",
    "WeightTraining": "red",
    "Yoga": "grey",
    "Swim": "blue",
    "Hike": "green",
}


fig, ax = plt.subplots()

scatter = ax.scatter(
    hours,
    cool_data["distance"],
    color=[activity_colors[i] for i in cool_data["type"]],
)

# Create handles for legend
legend_handles = []
for activity, color in activity_colors.items():
    legend_handles.append(
        plt.Line2D(
            [0],
            [0],
            marker="o",
            color="w",
            markerfacecolor=color,
            markersize=10,
            label=activity,
        )
    )

# Set legend
ax.legend(handles=legend_handles, fontsize=8)


# Labelling
plt.title("Time vs. Distance")
plt.xlabel("Time (Hr)")
plt.ylabel("Distance (m)")


# if the run is over 10 miles, then it gets a label!
for i, run in enumerate(cool_data["distance"]):
